reModal 模态窗口

基于jquery.js开发的模态窗口组件

View on GitHub
默认窗口 改变宽度 改变高度 改变宽和高 改变位置 不重置表单 遮罩不可点击关闭 改变弹框背景色 改变遮罩背景色 改变关闭按钮背景色 错误提示 正确提示 显示加载 移除加载 显示前回调 隐藏后回调 淡出关闭效果 鼠标移入触发 不显示关闭按钮
  • 默认弹出层

    'box'配置项表示弹出层的class 或者id 

    $('.demo0').reModal({box:'#reModalBox0'});
  • 改变宽度 
    $('.demo1').reModal({box:'#reModalBox1',width: 500});
  • 改变高度 
    $('.demo2').reModal({box:'#reModalBox2',height: 400});
  • 改变宽和高 
    $('.demo3').reModal({box:'#reModalBox3',width:600,height: 500});
  • 改变位置

    'positions'表示弹出层的位置，默认为center居中，可配置成 top/left/center

    $('.demo4').reModal({box:'#reModalBox4',positions: 'top'});
  • 不重置表单

    关闭弹出层时，默认将弹出层中的表单置空，配置'resetForm'为false可以实现表单不置空

    $('.demo5').reModal({box:'#reModalBox5',resetForm: false});
  • 遮罩不可点击关闭

    默认遮罩层点击可关闭

    $('.demo6').reModal({box:'#reModalBox0',modalHide: false});
  • 改变弹框背景色 
    $('.demo7').reModal({box:'#reModalBox0',boxBg: '#eeeeee'});
  • 改变遮罩背景色 
    $('.demo8').reModal({box:'#reModalBox0',modalBg: 'rgba(255,255,255,0.7)'});
  • 改变关闭按钮背景色 
    $('.demo9').reModal({box:'#reModalBox0',closeBg: '#BC2600'});
  • 错误提示

    $.tooltip('t1',t2,t3);
    * @param t1 : string 提示文字
    * @param t2 : string 提示类型，成功为success 警告为alert,默认为success
    * @param t3 : string 提示时间，默认事件为3000 

    $('.demo10').click(function(){
  $.tooltip('My God, 出错啦！！！','alert');
});
  • 正确提示 
    $('.demo11').click(function(){
  $.tooltip('OK, 操作成功！');
});
  • 显示加载 
    $('.demo12').click(function(){
       //自定义文字: $.showLoading('玩命加载中...');
      //自定义宽高: $.showLoading('努力加载中...',140,40);
      $.showLoading();
});
  • 移除加载 
     $('.demo13').click(function(){
   $.hideLoading();
});
  • 显示前回调 
    $('.demo14').reModal({box:'#reModalBox0',beforeShow: function(){
   alert('显示前执行');
}});
  • 隐藏后回调 
    $('.demo15').reModal({box:'#reModalBox0',afterHide: function(){
  alert('隐藏后执行');
}});
  • 淡出关闭效果 
    $('.demo16').reModal({box:'#reModalBox0',effect: 'fadeOut'});
  • 鼠标移入触发 
    $('.demo17').reModal({box:'#reModalBox0',triggerEvent: 'mouseenter'});
  • 不显示关闭按钮 
    $('.demo18').reModal({box:'#reModalBox0',closeHide: false});
            $('.demo0').reModal({
               box:'#reModalBox0',//弹出层id或者class绑定
               width: 480,//设置宽度
               height: '',//设置高度
               positions: 'center',//弹出层显示位置  top/left/center(默认)
               resetForm: true,//关闭弹出层后是否重置表单 默认为true
               modalHide: false,//遮罩成是否可以点击关闭，默认为ture
               boxBg: '#eeeeee',//设置弹出层内容区域的北背景色
               modalBg: 'rgba(255,255,255,0.7)',//遮罩层的背景色
               closeBg: '#BC2600',//改变关闭按钮背景色,
               closeHide: false,//不显示关闭按钮
               triggerEvent: 'mouseenter',//鼠标移入触发
               effect: 'fadeOut',//淡出关闭效果
               escHide:'true',Esc键关闭弹出层
               afterHide: function(){//隐藏后回调
                  alert('隐藏后执行');
               },
               beforeShow: function(){//显示前回调
                  alert('显示前执行');
               }
            });
